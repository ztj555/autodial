    <script>
        // 切换开关
        function toggleSwitch(btn) {
            const thumb = btn.querySelector('.toggle-thumb');
            const isOn = btn.classList.contains('bg-primary');
            if (isOn) {
                btn.classList.remove('bg-primary');
                btn.classList.add('bg-border');
                thumb.classList.remove('translate-x-5');
            } else {
                btn.classList.remove('bg-border');
                btn.classList.add('bg-primary');
                thumb.classList.add('translate-x-5');
            }
        }

        // 切换页面
        function switchTab(tabName) {
            // 隐藏所有页面
            document.querySelectorAll('[id^="page-"]').forEach(page => {
                page.classList.remove('page-active');
                page.classList.add('page-hidden');
            });
            
            // 显示目标页面
            const targetPage = document.getElementById('page-' + tabName);
            if (targetPage) {
                targetPage.classList.remove('page-hidden');
                targetPage.classList.add('page-active');
            }
            
            // 更新导航状态
            document.querySelectorAll('[id^="nav-"]').forEach(nav => {
                nav.classList.remove('nav-item-active');
                nav.classList.add('nav-item-inactive');
            });
            const activeNav = document.getElementById('nav-' + tabName);
            if (activeNav) {
                activeNav.classList.remove('nav-item-inactive');
                activeNav.classList.add('nav-item-active');
            }
        }

        // 登记按钮状态切换
        function showSubmitState(state) {
            const states = ['default', 'loading', 'success', 'error'];
            states.forEach(s => {
                const btn = document.getElementById('btn-submit-' + s);
                if (btn) {
                    btn.classList.add('hidden');
                }
            });
            const targetBtn = document.getElementById('btn-submit-' + state);
            if (targetBtn) {
                targetBtn.classList.remove('hidden');
            }
        }

        // 初始化
        document.addEventListener('DOMContentLoaded', function() {
            // 默认显示连接页
            switchTab('connect');
        });
    </script>
